// feature() - compile-time gate function

export function feature(name: string): boolean {

  // External builds: always returns false

  // Internal builds: checks GrowthBook

  if (!isInternalBuild()) return false;

  return checkGate(name);

}

 

// Usage pattern - positive gating

if (feature('VOICE_MODE')) {

  // This entire block is DCE'd

  // in external builds

  enableVoiceMode();

}

Dead Code Elimination (DCE)

For external builds, feature() is statically known to return false. Bundler (esbuild) eliminates all gated code blocks at compile time. Zero runtime overhead.

Positive Gating Pattern

  • Always gate on feature presence, not absence

  • if (feature('X')) - correct

  • if (!feature('X')) - anti-pattern (blocks DCE)

  • Ensures clean removal of experimental code

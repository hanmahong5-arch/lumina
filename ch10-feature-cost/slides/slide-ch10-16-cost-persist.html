// saveCurrentSessionCosts()

async function saveCurrentSessionCosts() {

  const costState = getCurrentCostState();

  await writeProjectConfig(sessionId, {

    costs: {

      totalUSD: costState.totalSessionCost,

      modelUsage: costState.modelUsage,

      lastUpdated: Date.now(),

    }

  });

}

Session Resume

When resuming a conversation with --resume, the persisted cost state is loaded and the accumulator continues from where it left off. This gives accurate total cost for long-running sessions.

// getMergedBetas() logic

function getMergedBetas(model, provider) {

  let betas = [...BASE_BETAS];

  

  // Add model-specific betas

  if (supportsThinking(model)) {

    betas.push('INTERLEAVED_THINKING');

  }

  

  // Provider-specific exclusions

  if (provider === 'bedrock') {

    betas = betas.filter(b =>

      !BEDROCK_UNSUPPORTED.has(b));

  }

  

  // Feature-gated betas

  if (feature('CITATIONS_BETA')) {

    betas.push('CITATIONS');

  }

  

  return [...new Set(betas)];

}

Bedrock/Vertex Special Handling

  • Bedrock: No PROMPT_CACHING beta (native support)

  • Vertex: Different beta header format

  • Some betas are API-provider specific

  • Provider detection at request time

Merge Order Matters

  • Base betas applied first (always-on)

  • Model betas add capabilities

  • Provider filter removes incompatible

  • Feature gates add experimental

  • Final dedup prevents header duplication

// core/src/session/handlers.rs:962

pub(super) async fn submission_loop(

sess: Arc<Session>,

config: Arc<Config>,

rx_sub: Receiver<Submission>,

) {

while let Ok(sub) = rx_sub.recv().await {

let span = submission_dispatch_span(&sub);

let should_exit = async {

match sub.op.clone() {

Op::Interrupt => { interrupt(&sess).await; false }

Op::UserInput { items } => { ... false }

Op::Shutdown => { ...; true }

// ~20 种 Op variant

}

}.instrument(span).await;

if should_exit { break; }

}

}

骨架 1：纯顺序 while 循环，无 tokio::select。每次从 rx_sub 拉一个 Submission，处理完再拉下一个。

骨架 2：根据 sub.op enum variant 分派到对应处理函数——约 20 种 Op，每种独立处理路径。

骨架 3：每个 dispatch 单独 instrument 一个 tracing span——可观测性内置，不是事后添加。

骨架 4：收到 Op::Shutdown 返回 true，退出循环。优雅关闭走同一消息通道。

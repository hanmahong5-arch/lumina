指数退避 / Exponential Backoff

BASE_DELAY=500ms, max=32s, 25% jitter, Retry-After priority

function getBackoffDelay(attempt, error) {

// Retry-After header takes priority

const retryAfter = error.headers

?.['retry-after'];

if (retryAfter) return retryAfter * 1000;

const base = 500; // BASE_DELAY

const exp = base * Math.pow(2, attempt);

const max = 32000; // cap

const capped = Math.min(exp, max);

// 25% jitter

return capped * (0.75 + Math.random() * 0.5);

}
